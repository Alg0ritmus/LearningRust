// select number of threads used while testing

$ cargo test -- --test-threads=1

// by default if fn that we test is printing smth, 
// it prints msg only if test fail, but we can change
// that behavior in console, so println! will be captured
// even if test passes

$ cargo test -- --show-output

// running single test

$ cargo test "testname"  // u can specify just substring and it will filter
all matching tests

// u can exclude time-consuming test from running like so:

#[test]
#[ignore]


// running #[ignore] tests:

$ cargo test -- --ignored

// run all tests + ignored tests

$ cargo test -- --include-ignored




// run specific integration test

$ cargo test --test "integration_test_filename"
